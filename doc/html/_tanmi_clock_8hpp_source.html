782 inline bool Clock::isUpdate(const ClockID _id)
783 {
784 std::shared_ptr<ClockElem> i(nullptr);
785 try
786 {
787 i = getIterator(_id);
788 if (i.get() == nullptr)
789 throw ClockNotFoundException();
790
791 if (i->pause == true)
792 {
793 return false;
794 }
795 std::lock_guard<std::mutex> lock(i->templock);
796 QueryPerformanceCounter(&i->temp_lint);
797 if (i->temp_lint.QuadPart - i->cycle > i->update_tick)
798 {
799 i->last_cycle = i->cycle;
800 i->cycle = i->temp_lint.QuadPart;
801 auto relative_passed = (i->cycle - i->last_cycle) * i->scale;
802 i->relative_tick += relative_passed;
803#ifdef EVENT_SYSTEM
804 EventSystem& eventSystem = EventSystem::Instance();
805 QueryPerformanceFrequency(&i->temp_lint);
806 auto freq = i->temp_lint.QuadPart;
807 std::lock_guard<std::mutex> lock(i->lock);
808 for (auto e : i->eventList)
809 {
810 eventSystem.TriggerEventUpdate(e, relative_passed * 1000 / freq);
811 }
812#endif // EVENT_SYSTEM
813 return true;
814 }
815 else
816 {
817 return false;
818 }
819 }
820 catch (ClockException& exp)
821 {
822 std::cout << "\n::Clock::isUpdate()" << exp.what() << std::endl;
823 }
824 return false;
