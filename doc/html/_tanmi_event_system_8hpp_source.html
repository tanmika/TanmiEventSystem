427 inline void EventSystem::TriggerEvent(const Event& event)
428 {
429 try
430 {
431 // 寻找事件
432 auto range = EventList.equal_range(event.ID);
433 if (std::distance(range.first, range.second) == 0)
434 {
435 throw EventSystemEventNotFoundException();
436 }
437 // 预处理
438 if (eventsPreprocess[event.ID]() == false)
439 return;
440 // 定位
441 auto& it = range.first;
442 auto& it_end = range.second;
443 std::lock_guard<std::mutex> lock(mtx_temp);
444 // 压缓冲
445 while (it != it_end)
446 {
447 if (!it->second.expired())
448 {
449 listenersToWake.push_back(it->second.lock());
450 it++;
451 }
452 else
453 {
454 EventList.erase(it++);
455 }
456 }
457 // 通知
458 for (auto& listener : listenersToWake)
459 {
460 messageHandler->Post(event.ID, listener);
461 }
462 // 清空缓冲
463 listenersToWake.clear();
464 }
465 catch (EventSystemException& e)
466 {
467 std::cout << "::EventSystem::WakeEvent()" << e.what() << std::endl;
468 }
